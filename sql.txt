drop table member;
drop table comments;
drop table post;
drop table post_member_good;

drop sequence post_seq;
drop sequence comments_seq;

create sequence post_seq;
create sequence comments_seq;

create table member (
	username varchar2(10 char),
	password varchar2(60 char) ,
	email varchar2(30 char),
	profile clob,
	join_day date,
	role varchar2(15 char),
	levels varchar2(15 char),
  failed_attempts number(1),
	is_lock number(1),
	constraint member_pk_username primary key(username)
);

create table post (
	pno number(6),
	title varchar2(50 char),
	content clob,
	writer varchar2(10 char),
	write_time date,
	read_cnt number(6),
	good_cnt number(6),
	bad_cnt number(6),
	constraint post_pk_pno primary key(pno)
);

create table comments (
	cno number(6),
	content varchar2(100 char),
	writer varchar2(10 char),
	write_time date,
	pno number(6),
	constraint comments_pk_cno primary key(cno),
  constraint comments_fk_pno foreign key(pno) references post(pno)
);

create table post_member_good (
	pno number(6),
	username varchar2(10 char),
	constraint post_member_good_pk primary key(pno, username)
);

create table post_member_bad (
	pno number(6),
	username varchar2(10 char),
	constraint post_member_bad_pk primary key(pno, username)
);
